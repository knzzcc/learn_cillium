# CKA 指令速查

## Context 切換（每道題必做）

```bash
k config use-context <context-name>
kn <namespace>
k config current-context
```

## Pod

```bash
k run nginx --image=nginx:1.27
k run nginx --image=nginx:1.27 --port=80 --labels=app=web
k run nginx --image=nginx:1.27 $do > pod.yaml
k delete pod nginx $now
```

## Deployment

```bash
k create deployment webapp --image=nginx:1.27 --replicas=3
k scale deployment webapp --replicas=5
k set image deployment/webapp nginx=nginx:1.28
k rollout status deployment/webapp
k rollout history deployment/webapp
k rollout undo deployment/webapp
k rollout undo deployment/webapp --to-revision=2
```

## Service

```bash
k expose deployment webapp --port=80 --target-port=80 --type=ClusterIP
k expose deployment webapp --port=80 --target-port=80 --type=NodePort
k port-forward svc/webapp 8080:80
```

## RBAC

```bash
k create sa my-sa -n dev
k create role pod-reader --verb=get,list,watch --resource=pods -n dev
k create rolebinding read-pods --role=pod-reader --serviceaccount=dev:my-sa -n dev
k create clusterrole node-reader --verb=get,list --resource=nodes
k create clusterrolebinding read-nodes --clusterrole=node-reader --serviceaccount=dev:my-sa

# 驗證權限
k auth can-i list pods -n dev --as=system:serviceaccount:dev:my-sa
```

## Node 管理

```bash
k cordon <node>
k drain <node> --ignore-daemonsets --delete-emptydir-data
k uncordon <node>
k label nodes <node> disk=ssd
k taint nodes <node> key=value:NoSchedule
k taint nodes <node> key=value:NoSchedule-    # 移除
```

## etcd Backup / Restore

```bash
# Backup
ETCDCTL_API=3 etcdctl snapshot save /tmp/etcd-backup.db \
  --endpoints=https://127.0.0.1:2379 \
  --cacert=/etc/kubernetes/pki/etcd/ca.crt \
  --cert=/etc/kubernetes/pki/etcd/server.crt \
  --key=/etc/kubernetes/pki/etcd/server.key

# 驗證
ETCDCTL_API=3 etcdctl snapshot status /tmp/etcd-backup.db --write-table

# Restore
ETCDCTL_API=3 etcdctl snapshot restore /tmp/etcd-backup.db \
  --data-dir=/var/lib/etcd-restored

# 修改 etcd manifest
sudo vi /etc/kubernetes/manifests/etcd.yaml
# 改 --data-dir 與 hostPath.path 為 /var/lib/etcd-restored
```

> cert 路徑從 `cat /etc/kubernetes/manifests/etcd.yaml` 確認

## Kubeadm Upgrade

```bash
# Control plane
sudo apt-mark unhold kubeadm && sudo apt-get install -y kubeadm=1.35.0-1.1
sudo kubeadm upgrade plan
sudo kubeadm upgrade apply v1.35.0
sudo apt-get install -y kubelet=1.35.0-1.1 kubectl=1.35.0-1.1
sudo systemctl daemon-reload && sudo systemctl restart kubelet

# Worker（先從 control plane drain）
k drain worker-1 --ignore-daemonsets --delete-emptydir-data
# ssh 進 worker 執行
sudo kubeadm upgrade node
sudo systemctl daemon-reload && sudo systemctl restart kubelet
# 回 control plane
k uncordon worker-1
```

## Troubleshooting 快速指令

```bash
# Pod 問題
k describe pod <pod>
k logs <pod>
k logs <pod> --previous
k get events --sort-by='.lastTimestamp'

# Node 問題
k get nodes
ssh <node> -- sudo systemctl status kubelet
ssh <node> -- sudo journalctl -u kubelet --no-pager | tail -50

# Service 問題
k get endpoints <svc>
k describe svc <svc>

# DNS 測試
k run test --image=busybox:1.36 --rm -it -- nslookup kubernetes
```
