# CKS 指令速查

## 考試第一分鐘設定

```bash
alias k=kubectl
alias kn='kubectl config set-context --current --namespace'
export do='--dry-run=client -o yaml'
export now='--force --grace-period=0'
source <(kubectl completion bash)
complete -o default -F __start_kubectl k
export ETCDCTL_API=3

cat <<'EOF' >> ~/.vimrc
set expandtab
set tabstop=2
set shiftwidth=2
set number
EOF
```

## Context / Namespace（每題必做）

```bash
k config use-context <context-name>
kn <namespace>
k config current-context
```

## RBAC

```bash
k create sa my-sa -n my-ns
k create role pod-reader --verb=get,list,watch --resource=pods -n my-ns
k create rolebinding read-pods --role=pod-reader --serviceaccount=my-ns:my-sa -n my-ns
k create clusterrole node-reader --verb=get,list --resource=nodes
k create clusterrolebinding read-nodes --clusterrole=node-reader --serviceaccount=my-ns:my-sa

# 驗證
k auth can-i list pods -n my-ns --as=system:serviceaccount:my-ns:my-sa
k auth can-i --list --as=system:serviceaccount:default:default

# 找 cluster-admin 過度授權
k get clusterrolebinding -o wide | grep cluster-admin
```

## NetworkPolicy

```bash
k get networkpolicy -n <ns>
k describe networkpolicy <name> -n <ns>

# 測試 pod 之間連線
k exec <pod-a> -- wget -qO- --timeout=2 http://<pod-b-ip>:<port>
k exec <pod-a> -- nc -zv <pod-b-ip> <port>
```

## Pod Security Admission

```bash
# 套用 namespace label
k label ns <ns> pod-security.kubernetes.io/enforce=restricted
k label ns <ns> pod-security.kubernetes.io/audit=restricted
k label ns <ns> pod-security.kubernetes.io/warn=restricted

# 確認
k get ns <ns> --show-labels
```

## Secrets

```bash
k create secret generic my-secret --from-literal=key=value
k create secret tls tls-secret --cert=tls.crt --key=tls.key

# 查看解碼後的值
k get secret my-secret -o jsonpath='{.data.key}' | base64 -d

# 確認加密靜態資料是否啟用
grep encryption-provider-config /etc/kubernetes/manifests/kube-apiserver.yaml
```

## Image Scanning（Trivy）

```bash
trivy image nginx:1.27
trivy image --severity CRITICAL,HIGH nginx:1.27
trivy image --severity CRITICAL nginx:1.27 --quiet
trivy image --severity CRITICAL nginx:1.27 -o results.txt
trivy config Dockerfile
```

## Audit Logging

```bash
# 確認是否啟用
grep audit /etc/kubernetes/manifests/kube-apiserver.yaml

# 查看 log
cat /var/log/kubernetes/audit/audit.log | tail -20
cat /var/log/kubernetes/audit/audit.log | jq 'select(.verb=="create")'
cat /var/log/kubernetes/audit/audit.log | jq 'select(.user.username=="<username>")'
```

## Falco

```bash
systemctl status falco
sudo systemctl restart falco
journalctl -u falco --no-pager | tail -30
cat /var/log/syslog | grep falco

# 規則位置
ls /etc/falco/
cat /etc/falco/falco_rules.yaml
ls /etc/falco/rules.d/
```

## AppArmor

```bash
# 確認 profile 狀態
sudo aa-status
sudo apparmor_status

# 載入 profile
sudo apparmor_parser -q /etc/apparmor.d/my-profile

# 確認已載入
sudo aa-status | grep profiles
```

## kube-bench（CIS Benchmark）

```bash
kube-bench run --targets=master
kube-bench run --targets=node
kube-bench run --targets=etcd
kube-bench run --targets=master --check=1.2.16
```

## 容器除錯

```bash
k exec <pod> -- ps aux
k exec <pod> -- netstat -tlnp
k exec <pod> -- cat /proc/1/cmdline

# crictl（在 node 上）
sudo crictl ps
sudo crictl inspect <container-id>
sudo crictl logs <container-id>

# kubectl debug
k debug pod/<pod-name> -it --image=busybox:1.36
k debug node/<node-name> -it --image=busybox:1.36
```
